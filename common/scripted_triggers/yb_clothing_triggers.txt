yb_clothing_culture_trigger = {
	culture = {
		OR = {
			this = culture:$CULTURE_FLAG$
			any_parent_culture_or_above = { this = culture:$CULTURE_FLAG$ }
		}
	}
}

portrait_yb_clothing_gfx_trigger = {
	trigger_if = {
		limit = { exists = this	}
		scope:culture = { has_clothing_gfx = $CULTURE_FLAG$_clothing_gfx }
	}
	trigger_else = {
		scope:culture = culture:$CULTURE_FLAG$
	}
}

portrait_yb_muslim_clothing_trigger = {
	scope:faith.religion = religion:islam_religion
}
